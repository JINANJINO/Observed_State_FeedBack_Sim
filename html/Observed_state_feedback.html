clc; clear; close all;

% Plant
A = [0 1; 0 0];
B = [0; 1];
C = [1 0];
D = 0;

% Gains (use place for better numerical stability)
K = place(A, B, [-1 -1]);             % controller poles at -1 (double)
L = place(A', C', [-5 -5])';          % observer poles faster (here at -5)

% Augmented closed-loop + observer system (states: x(2) and xhat(2) -> 4 states)
Acl = [A, -B*K;
       L*C, A - B*K - L*C];

Bcl = [B; zeros(2,1)];                % single input (can be reference or control input)
Ccl = [C, zeros(1,2)];
Dcl = 0;

h = ss(Acl, Bcl, Ccl, Dcl);

t = 0:0.01:8;

% Example 1: autonomous response (no input) from IC (x1=1, x2=0.5, xhat=0)
x0 = [1; 0.5; 0; 0];
u_zero = zeros(size(t));
[y1, t1, xhist1] = lsim(h, u_zero, t, x0);

% Example 2: apply a unit step input (just to see forced response) -- optional
u_step = ones(size(t));
[y2, t2, xhist2] = lsim(h, u_step, t, x0);

% Plot autonomous response
figure;
subplot(2,1,1)
plot(t1, xhist1(:,1), '-b', 'LineWidth', 2); hold on
plot(t1, xhist1(:,3), '--r', 'LineWidth', 2);
axis([0 8 -0.5 1.5])
xlabel('Time (sec)'); ylabel('State x_1 / \hat{x}_1')
legend('x_1 (true)', '\hat{x}_1 (est)', 'Location', 'Best')
grid on
set(gca, 'FontName', 'times', 'FontSize', 14)

subplot(2,1,2)
plot(t1, xhist1(:,2), '-b', 'LineWidth', 2); hold on
plot(t1, xhist1(:,4), '--r', 'LineWidth', 2);
axis([0 8 -1 1])
xlabel('Time (sec)'); ylabel('State x_2 / \hat{x}_2')
legend('x_2 (true)', '\hat{x}_2 (est)', 'Location', 'Best')
grid on
set(gca, 'FontName', 'times', 'FontSize', 14)
